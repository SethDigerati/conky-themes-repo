# Overload Conky Theme Dependencies

## System Packages (Arch Linux)
conky              # Conky system monitor with Lua support
lua                # Lua interpreter
curl               # For API requests (weather, network)
imlib2             # Image loading library for icons

## Optional Packages
jq                 # JSON processor for debugging
lm-sensors         # Hardware monitoring (temperature, fan speed)
intel-gpu-tools    # Intel GPU monitoring (optional)
speedtest-cli      # Internet speed test tool

## Fonts (for weather widget)
ttf-dejavu         # DejaVu fonts (wind direction arrows)
noto-fonts-emoji   # Noto Color Emoji (moon phase icons)

## Installation Commands (Arch Linux):
# pacman -S conky lua curl imlib2 ttf-dejavu noto-fonts-emoji

## Installation Commands (Ubuntu/Debian):
# sudo apt install conky-all lua5.3 curl libimlib2 fonts-dejavu fonts-noto-color-emoji

## Weather Widget Setup:
# 1. Copy weather_api_config.lua.template to weather_api_config.lua
# 2. Get a free API key from https://openweathermap.org/api
# 3. Replace YOUR_API_KEY_HERE with your actual API key
# 4. Run: conky -c ~/.conky/conky-themes-repo/overload/weather.conf
